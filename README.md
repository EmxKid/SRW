# запуск с указанием параметров
./simulator --users 10 --time 1e6 --seed 42 \
            --active "exp:0.5" \
            --passive "exp:0.3333" \
            --csv "output.csv"

# Показать справку
./simulator --help

| Параметр   | Краткий | Тип      | По умолчанию   | Описание                          |
|------------|---------|----------|----------------|-----------------------------------|
| `--users`  | `-u`    | `int`    | `10`           | Количество пользователей          |
| `--time`   | `-t`    | `double` | `1e6`          | Время симуляции (сек)             |
| `--seed`   | `-s`    | `int`    | `42`           | Seed ГСЧ (воспроизводимость)      |
| `--active` | `-a`    | `string` | `"exp:0.5"`    | Распределение активной фазы       |
| `--passive`| `-p`    | `string` | `"exp:0.3333"` | Распределение пассивной фазы      |
| `--csv`    | `-o`    | `string` | —              | Файл для сохранения P(k)          |
| `--help`   | `-h`    | —        | —              | Показать справку                  |


| Тип             | Синтаксис              | Параметры       | Ограничения      | E[X]               |
|-----------------|------------------------|-----------------|------------------|--------------------|
| Экспоненциальное| `exp:rate`             | `rate`          | `rate > 0`       | `1/rate`           |
| Детерминированное| `det:value`           | `value`         | —                | `value`            |
| Нормальное      | `norm:mean,std`        | `mean, std`     | `std > 0`        | `mean`             |
| Гамма           | `gamma:shape,scale`    | `shape, scale`  | `> 0`            | `shape × scale`    |
| Логнормальное   | `lognorm:mu,sigma`     | `mu, sigma`     | `sigma > 0`      | `exp(μ + σ²/2)`    |
| Равномерное     | `unif:min,max`         | `min, max`      | `min < max`      | `(min+max)/2`      |


# Экспоненциальные фазы (базовая M/M/N/N)
./simulator --active "exp:0.5" --passive "exp:0.3333"

# Нормальное активное + детерминированное пассивное
./simulator --active "norm:2.0,0.5" --passive "det:3.0"

# Гамма + экспоненциальное с сохранением в CSV
./simulator --active "gamma:2,1" --passive "exp:0.3333" --csv result.csv

# Равномерное распределение
./simulator --active "unif:1.0,3.0" --passive "unif:2.0,4.0"
